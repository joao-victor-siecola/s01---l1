
% progenitor(PaiOuMae, Filho)
progenitor(cronos, zeus).
progenitor(cronos, poseidon).
progenitor(cronos, hades).

progenitor(reia, zeus).
progenitor(reia, poseidon).
progenitor(reia, hades).

% dominio(Deus, Dominio)
dominio(zeus, ceu).
dominio(zeus, relampago).

dominio(poseidon, mar).
dominio(poseidon, terremotos).

dominio(hades, submundo).
dominio(hades, riquezas).

% local(Deus, LocalOndeHabita)
local(zeus, olimpo).
local(poseidon, mar).
local(hades, submundo).

divindade_olimpica(Deus) :-
    progenitor(cronos, Deus),
    dominio(Deus, D),
    (D = ceu ; D = mar ; D = submundo).

% Auxiliar: verifica se tem pelo menos dois dominios distintos
dois_dominios(Deus) :-
    dominio(Deus, D1),
    dominio(Deus, D2),
    D1 \= D2.

deus_maior(Deus) :-
    dois_dominios(Deus),
    local(Deus, olimpo).

irmaos_germanos(A, B) :-
    progenitor(P, A), progenitor(P, B),
    progenitor(M, A), progenitor(M, B),
    P \= M,
    A \= B.

% caso base
ancestral(A, D) :-
    progenitor(A, D).

% passo recursivo
ancestral(A, D) :-
    progenitor(A, Z),
    ancestral(Z, D).

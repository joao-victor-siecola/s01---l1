% Locais principais
local_principal(zeus, olimpo).

divindade_olimpica(Deus) :-
    progenitor(cronos, Deus),
    dominio(Deus, D),
    (D = ceu ; D = mar ; D = submundo; D = florestas; D = atlantida).

% Auxiliar: verifica se tem pelo menos dois dominios distintos
tem_dois_dominios(Deus) :-
    dominio(Deus, D1),
    dominio(Deus, D2),
    D1 \= D2.

deus_maior(Deus) :-
    tem_dois_dominios(Deus),
    local_principal(Deus, olimpo).

irmaos_germanos(A, B) :-
    progenitor(P, A), progenitor(P, B),
    progenitor(M, A), progenitor(M, B),
    P \= M,
    A \= B.

% caso base
ancestral(A, D) :-
    progenitor(A, D).

% passo recursivo
ancestral(A, D) :-
    progenitor(A, Z),
    ancestral(Z, D).
